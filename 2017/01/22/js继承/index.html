js继承实现方法

Posted on 2017-01-22 | In JavaScript

    <title></title>

<script type="text/javascript">
//原型链继承
function SupType(){
    this.name = "jack";
    this.color=["red","green"];
}
function SubType(){
    this.age="8";
}
SubType.prototype = new SupType();
SubType.prototype.showName = function(){
    return this.name;
}
//缺点：对于引用类型值 实例之间会互相影响

//构造函数继承
function SupType(name){
    this.name = name;
    this.color=["red","green"];
    this.sayName = function(){
        return this.name;
    }
}
function SubType(name){
    SupType.call(this,name);
}
//优点：引用类型值 实例之间不会互相影响 可以传参 
//缺点：没实现方法复用

//组合继承
function SupType(name){
    this.name = name;
    this.color=["red","green"];
}
function SubType(name,age){
    SupType.call(this,name);
    this.age = age;
}
SubType.prototype = new SupType();
SubType.prototype.sayName = function(){
    return this.name;
}
//使用构造函数继承属性，使用原型链继承方法 既属性互不影响，又实现了方法复用 
//缺点：使用了两次超类构造函数

//寄生组合式继承
function SupType(name){
    this.name = name;
    this.color=["red","green"];
}
function SubType(name,age){
    SupType.call(this,name);
    this.age = age;
}
//SubType.prototype = new SupType();
inheritPrototype(SubType,SupType);
//把超类实例赋给子类的原型。
function inheritPrototype(SubType,SupType){
    var proto = object(SubType.prototype);
    proto.constructor = SubType;
    SubType.prototype = proto;
}
// 返回一个以o为原型的实例
function object(o){
 function F(){};
 F.prototype = o;
 return new F();
}
SubType.prototype.sayName = function(){
    return this.name;
}
//只使用了一次超类的构造函数
</script>
