<script type="text/javascript">
function BinarySearchTree(){
    var Node = function(key){
        this.key = key;
        this.left = null;//左子节点
        this.right = null;//右子节点
    };//表示树中的节点
    var root = null; //表示根节点
    this.insert = function(key){
        var newNode = new Node(key);
        if (root ===null) {
            root = newNode;
        }else{
            insertNode(root,newNode);
        }
    };
    var insertNode = function(node,newNode){
        if (node.key>newNode.key) {
            if (node.left===null) {
                node.left = newNode;
            }else{
                //递归调用本身
                insertNode(node.left,newNode);
            }
        }else{
            if (node.right ===null) {
                node.right = newNode;
            }else{
                insertNode(node.right,newNode);
            }
        }
    };
    //查找最小值 
    this.min = function(){
        return minNode(root);
    }
    var minNode = function(node){
        if (node) {
            while(node&&node.left!==null){
                node = node.left; //找最大值同理 变成遍历右子树就好了
            }
            return node.key;
        }
        return null;
    };
    //查找一个特定的值是否存在
    this.search = function(key){
        return searchNode(root,key);
    }
    var searchNode = function(node,key){
        if (node===null) {
            return false;
        }
        if (key<node.key) {
            return searchNode(node.left,key);
        }else if(key>node.key){
            return searchNode(node.right,key);
        }else{
            return true;
        }
    };

}
</script>

<script type="text/javascript">
    function Set(){
        var items = {};
        //判断是否含有某个值
        this.has = function(value){
            return items.hasOwnProperty(value);
        };
        //新增
        this.add = function(value){
            if (!this.has(value)) {
                items[value] = value;
                return true;
            }
            return false;
        };
        //删除
        this.remove = function(value){
            if (this.has(value)) {
                delete items[value];
                return true;
            }
            return false;
        };
        //清除
        this.clear = function(){
            items = {};
        }
        //条数
        this.sizeLegacy = function(){
            var count = 0;
            for(var prop in items) {
                if(items.hasOwnProperty(prop)){ 
                ++count; //{7}
                }
                return count;
            }
        };
        //取出所有值
        this.valuesLegacy = function(){
            var keys = [];
            for(var key in items){ //{7}
            keys.push(key); //{8}
            }
            return keys;
        };
        //求并集
        this.union = function(otherSet){
            var unionSet = {};
            var values = this.values();
            for (var i = 0; i < values.length; i++) {
                unionSet.add(values[i]);
            }
            values = otherSet.values();
            for (var i = 0; i < values.length; i++) {
                unionSet.add(values[i]);
            }
            return unionSet;
        };
        //求交集
        this.intersection = function(otherSet){
            var intersectionSet = {};
            var values = otherSet.values();
            for (var i = 0; i < values.length; i++) {
                if (this.has(values[i])) {
                    intersectionSet.add(values[i]);
                }
            }
            return intersectionSet;
        };
        //求差集
        this.difference = function(otherSet){
            var differenceSet = {};
            var values = this.values();
            for (var i = 0; i < values.length; i++) {
                if (!otherSet.has(values[i])) {
                    intersectionSet.add(values[i]);
                }
            }
            return differenceSet;
        };
        //判断是否为其他集合的子集
        this.subset = function(otherSet){
            if (this.size()>otherSet.size()) {
                return false;
            }else{
                var values = this.values();
                for (var i = 0; i < values.length; i++) {
                    if (!otherSet.has(values[i])) {
                        return false;
                    }
                }
                return true;
            }
        };
    }

<script type="text/javascript">
function LinkedList() {
    var Node = function(element){ 
        this.element = element;
        this.next = null;
    };
    var length = 0; 
    //head 变量作为第一个节点的引用,是 LinkedList 类的私有变量（这意味着它不能在 LinkedList 实例外部被访问和更改，只有通过 LinkedList 实例才可以）。但是，如果我们需要在类的实现外部循环访问列表，
    //就需要提供一种获取类的第一个元素的方法。
    var head = null;  
    //向列表尾部添加一个新的项
    this.append = function(element){
        var node = new Node(element),current;
        if(head === null){ //如果head是null 说明原列表中是没有元素的
            head = node; //将node作为head 就有了第一个值。
        }else{
            //我们只有第一个元素的引用，所以需要遍历所有元素 当某一个元素的next是null时，说明这个元素是最后一个元素
            current = head;
            while(current.next){
                current = current.next;
            }
            //找到最后一个元素后，将最后一个元素的next指向node
            current.next = node;
        }
        //修改链表长度
        length++;
    };
    //从列表中移除一项。
    this.removeAt = function(position){
        if(position>-1 && position <length){
            var current = head;
            var previous,index = 0 ;
            //移除第一项
            if (position==0) {
                head  = current.next;
            }else{
                //遍历链表直到查到想要的那个值
                while(index ++ < position){
                    previous = current;
                    current = current.next;
                }
                //如果移除的不是第一项 那么要做的是 把要移除的前一个的next和自己的next链接起来。
                previous.next = current.next;
            }
            length--;
            return current.element;
        }
        //如果不是有效位置 返回null 意味着没有删除元素
        else{
            return null;
        }
    };
    //向列表的特定位置插入一个新的项。
    this.insert = function(position, element){
        if(position>=0 && position<=length){
            var node = new Node(element);
            var current = head,previous,index=0;
            //在第一个位置添加 直接把插入节点的next指向原head就可以了。
            if (position===0) {
                node.next = current;
                head = node;
            }else{
                //遍历查找相应位置
                while(index++<position){
                    previous = current;
                    current = current.next;
                }
                //current是想要插的位置，所以把插入节点的next指向原位置上的节点
                node.next = current;
                previous.next  = node;
            }
            length++;
            return true;
        }else{
            //位置越界
            return false;
        }
    };
    //返回元素在列表中的索引。如果列表中没有该元素则返回 -1 。
    this.indexOf = function(element){
        var current = head;
        var index = 0;
        while(current){
            if(current.element === element){
                return index;
            }
            index++;
            current = current.next;
        }
        return -1;
    };
    //根据元素值删除元素
    this.remove = function(element){
        var index = this.indexOf(element);
        return this.removeAt(index);
    }
    //如果链表中不包含任何元素，返回 true ，如果链表长度大于0则返回 false 。
    this.isEmpty = function() {
        return length===0;
    };
    //返回链表包含的元素个数。与数组的 length 属性类似。
    this.size = function() {
        return length;
    };
    //由于列表项使用了 Node 类，就需要重写继承自JavaScript对象默认的
    //toString 方法，让其只输出元素的值。
    this.toString = function(){
        var current = head;
        var string = '';
        while(current){
            string = string+current.element;
            current = current.next;
        }
        return string;
    };
}
var linkedlist = new LinkedList();
linkedlist.append("rrr");
linkedlist.append("yyy");
linkedlist.append("uuu");
linkedlist.removeAt(1);
linkedlist.insert(1,"insert");
linkedlist.remove("rrr");
console.log(linkedlist.toString()); //insertuuu
console.log(linkedlist.size());//2
console.log(linkedlist.isEmpty());//false
</script>

function LinkedList() {
    var Node = function(element){ 
        this.element = element;
        this.next = null;
        this.prev = null; //*新增
    };
    var length = 0; 
    //head 变量作为第一个节点的引用,是 LinkedList 类的私有变量（这意味着它不能在 LinkedList 实例外部被访问和更改，只有通过 LinkedList 实例才可以）。但是，如果我们需要在类的实现外部循环访问列表，
    //就需要提供一种获取类的第一个元素的方法。
    var head = null;  
    //用于保存列表最后一项
    var tail = null;
    //从列表中移除一项。
    this.removeAt = function(position){
        if(position>-1 && position <length){
            var current = head;
            var previous,index = 0 ;
            //如果是第一项
            if (position===0) {
                head = current.next;
                if (length==1) {
                    tail = null;
                }else{
                    head.prev = null;
                }
                //如果是最后一项
            }else if(position===length-1){
                current = tail;
                tail = current.prev;
                tail.next = null;
                //是其他位置
            }else{
                while(index++<position){
                    previous = current;
                    current = current.next;
                }
                //将前一个与下一个链接起来
                previous.next = current.next;
                current.next.prev = previous;
            }
            length--;
            return current.element;
        }
        //如果不是有效位置 返回null 意味着没有删除元素
        else{
            return null;
        }
    };
    //向双向链表的特定位置插入一个新的项。
    this.insert = function(position, element){
        if(position>=0 && position<=length){
            var node = new Node(element);
            var current = head,previous,index=0;
            //在起点插入
            if (position===0){
                if (!head) {
                    head = node;
                    tail = node;
                }else{
                    node.next = current;
                    current.prev = node;
                    head = node ;
                }
                //在结尾处插入
            }else if (position===length) {
                current = tail;
                current.next = node;
                node.prev = current;
                tail = node;
                //在其他位置插入
            }else{
                while(index++<position){
                    previous = current;
                    current = current.next;
                }
                node .next = current;
                previous.next = node;
                current.prev = node;
                node.prev = previous;
            }
            length++;
            return true;

        }else{
            //位置越界
            return false;
        }
    };
    //返回元素在列表中的索引。如果列表中没有该元素则返回 -1 。
    this.indexOf = function(element){
        var current = head;
        var index = 0;
        while(current){
            if(current.element === element){
                return index;
            }
            index++;
            current = current.next;
        }
        return -1;
    };
    //根据元素值删除元素
    this.remove = function(element){
        var index = this.indexOf(element);
        return this.removeAt(index);
    }
    //如果链表中不包含任何元素，返回 true ，如果链表长度大于0则返回 false 。
    this.isEmpty = function() {
        return length===0;
    };
    //返回链表包含的元素个数。与数组的 length 属性类似。

        return length;
    };

        }

    };
